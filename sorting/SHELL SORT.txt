#include<stdio.h>
void shell(int a[],int n)
{
    int gap,i,j;
    for(gap=n/2;gap>=1;gap=gap/2)
    {
        for(j=gap;j<n;j++)
        {
            for(i=j-gap;i>=0;i=i-gap)
            {
                if(a[i]<a[i+gap])
                {
                    break;
                }
                else
                {
                    int t=a[i];
                    a[i]=a[i+gap];
                    a[i+gap]=t;
                }
            }
        }
    }
}
int main()
{
    int n;
    scanf("%d",&n);
    int a[n];
    for(int i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    shell(a,n);
    for(int i=0;i<n;i++)
    {
        printf("%d ",a[i]);
    }
}
