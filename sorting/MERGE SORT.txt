#include<stdio.h>
void bubble(int a[],int n)
{
    for(int i=0;i<n-1;i++)
    {
        for(int j=0;j<n-1-i;j++)
        {
            if(a[j]>a[j+1])
            {
                int t=a[j];
                a[j]=a[j+1];
                a[j+1]=t;
            }
        }
    }
}
void merge(int a[],int beg,int end)
{
    int mid=(beg+end)/2;
    int n1=mid+1;
    int n2=end-mid;
    int l[n1],r[n2],a2[n1+n2];
    for(int i=0;i<n1;i++)
    {
        l[i]=a[i];
    }
    for(int i=0;i<n2;i++)
    {
        r[i]=a[mid+1+i];
    }
    bubble(l,n1);
    bubble(r,n2);
    int i=0,j=0,k=0;
    while(i<n1&&j<n2)
    {
        if(l[i]<r[j])
        {
            a2[k++]=l[i++];
        }
        else
        {
            a2[k++]=r[j++];
        }
    }
    while(i<n1)
    {
        a2[k++]=l[i++];
    }
    while(j<n2)
    {
        a2[k++]=r[j++];
    }
    for(int i=0;i<=end;i++)
    {
        printf("%d ",a2[i]);
    }
}
int main()
{
    int n;
    scanf("%d",&n);
    int a[n];
    for(int i=0;i<n;i++)
    {
        scanf("%d",&a[i]);
    }
    merge(a,0,n-1);
}
