// Java program for Russian Peasant Multiplication\
// Function to multiply two numbers using Russian Peasant method
	static int russianPeasant(int a, int b)
		int res = 0; 
		while (b > 0)
		{
			if ((b & 1) != 0)
				res = res + a;
			a = a << 1;
			b = b >> 1;
		}
		return res;
	}
